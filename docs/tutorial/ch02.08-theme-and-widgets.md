```rust
// New colors for Stage 2 widgets:
pub const SELECTION_BG: Color = Color::Rgb(69, 71, 90);       // highlighted row background
```

### Resource list widget

**crates/crystal-tui/src/widgets/resource_list.rs:**

```rust
use ratatui::prelude::*;
use ratatui::widgets::{Block, Borders, Cell, Paragraph, Row, Scrollbar, ScrollbarOrientation, ScrollbarState, Table};

use crate::theme;

// A table widget for displaying Kubernetes resources.
// Takes only borrowed data — no ownership of state, no dependency on crystal-app.
// This keeps crystal-tui completely decoupled from the application layer.
pub struct ResourceListWidget<'a> {
    pub title: &'a str,
    pub headers: &'a [String],
    pub items: &'a [Vec<String>],
    pub selected: Option<usize>,
    pub scroll_offset: usize,
    pub loading: bool,
    pub error: Option<&'a str>,
}

impl<'a> ResourceListWidget<'a> {
    pub fn render(self, frame: &mut Frame, area: Rect) {
        let block = Block::default()
            .borders(Borders::ALL)
            .border_style(Style::default().fg(theme::BORDER_COLOR))
            .title(format!(" {} ", self.title))
            .title_style(Style::default().fg(theme::ACCENT).bold());

        // Three special states rendered before the table:
        if self.loading {
            let msg = Paragraph::new("Loading...").style(Style::default().fg(theme::TEXT_DIM)).block(block);
            frame.render_widget(msg, area);
            return;
        }

        if let Some(err) = self.error {
            let msg =
                Paragraph::new(format!("Error: {err}")).style(Style::default().fg(theme::STATUS_FAILED)).block(block);
            frame.render_widget(msg, area);
            return;
        }

        if self.items.is_empty() {
            let msg = Paragraph::new("No resources found").style(Style::default().fg(theme::TEXT_DIM)).block(block);
            frame.render_widget(msg, area);
            return;
        }

        // Build column headers (bold blue)
        let header_cells: Vec<Cell> = self
            .headers
            .iter()
            .map(|h| Cell::from(h.as_str()).style(Style::default().fg(theme::TABLE_HEADER_FG).bold()))
            .collect();
        let header = Row::new(header_cells).height(1);

        // Build data rows. Column index 2 is the STATUS column — we color-code it.
        let rows: Vec<Row> = self
            .items
            .iter()
            .map(|item| {
                let cells: Vec<Cell> = item
                    .iter()
                    .enumerate()
                    .map(|(col_idx, val)| {
                        let style =
                            if col_idx == 2 { Style::default().fg(status_color(val)) } else { Style::default() };
                        Cell::from(val.as_str()).style(style)
                    })
                    .collect();
                Row::new(cells)
            })
            .collect();

        // Column widths: first column (NAME) gets more space.
        let widths: Vec<Constraint> = self
            .headers
            .iter()
            .enumerate()
            .map(|(i, _)| if i == 0 { Constraint::Min(20) } else { Constraint::Min(12) })
            .collect();

        let table = Table::new(rows, &widths)
            .header(header)
            .block(block)
            .row_highlight_style(Style::default().bg(theme::SELECTION_BG))
            .highlight_symbol("▶ ");

        // render_stateful_widget is ratatui's way of rendering widgets that have
        // internal state (like which row is selected). We create a TableState
        // and pass it alongside the widget.
        let mut table_state = ratatui::widgets::TableState::default().with_selected(self.selected);
        frame.render_stateful_widget(table, area, &mut table_state);

        // Show a scrollbar when items overflow the viewport.
        // area.height - 3 accounts for the border (2 lines) and header row (1 line).
        if self.items.len() > area.height.saturating_sub(3) as usize {
            let mut scrollbar_state = ScrollbarState::new(self.items.len()).position(self.selected.unwrap_or(0));
            let scrollbar =
                Scrollbar::new(ScrollbarOrientation::VerticalRight).style(Style::default().fg(theme::BORDER_COLOR));
            frame.render_stateful_widget(
                scrollbar,
                area.inner(Margin { vertical: 1, horizontal: 0 }),
                &mut scrollbar_state,
            );
        }
    }
}

// Maps a pod status string to a theme color.
fn status_color(status: &str) -> Color {
    match status {
        "Running" => theme::STATUS_RUNNING,
        "Succeeded" => theme::STATUS_RUNNING,
        "Pending" | "ContainerCreating" => theme::STATUS_PENDING,
        "Failed" | "Error" | "CrashLoopBackOff" | "ImagePullBackOff" => theme::STATUS_FAILED,
        _ => theme::STATUS_PENDING,
    }
}
```

### Namespace selector widget

**crates/crystal-tui/src/widgets/namespace_selector.rs:**

```rust
use ratatui::prelude::*;
use ratatui::widgets::{Block, Borders, Clear, List, ListItem, ListState, Paragraph};

use crate::theme;

// A popup overlay for switching namespaces.
// Renders as a centered box on top of the existing content.
// Supports type-to-filter: as the user types, the list narrows.
pub struct NamespaceSelectorWidget<'a> {
    pub namespaces: &'a [String],
    pub filter: &'a str,
    pub selected: usize,
}

impl<'a> NamespaceSelectorWidget<'a> {
    // Filter namespaces by the current filter string (case-insensitive contains).
    // "All Namespaces" is always included as the first entry (if it matches the filter).
    pub fn filtered_namespaces(&self) -> Vec<&'a str> {
        let mut result: Vec<&str> = Vec::new();
        let filter_lower = self.filter.to_lowercase();

        if filter_lower.is_empty() || "all namespaces".contains(&filter_lower) {
            result.push("All Namespaces");
        }

        for ns in self.namespaces {
            if filter_lower.is_empty() || ns.to_lowercase().contains(&filter_lower) {
                result.push(ns);
            }
        }

        result
    }

    pub fn render(self, frame: &mut Frame, area: Rect) {
        // Center the popup: 50% width, 60% height, capped at 60×30.
        let popup_width = area.width / 2;
        let popup_height = area.height * 3 / 5;
        let popup_area = Rect {
            x: area.x + (area.width.saturating_sub(popup_width)) / 2,
            y: area.y + (area.height.saturating_sub(popup_height)) / 2,
            width: popup_width.min(60),
            height: popup_height.min(30),
        };

        // Clear erases the area underneath — without this, the popup would
        // be rendered on top of the resource table and look garbled.
        frame.render_widget(Clear, popup_area);

        let block = Block::default()
            .borders(Borders::ALL)
            .border_style(Style::default().fg(theme::ACCENT))
            .title(" Select Namespace ")
            .title_style(Style::default().fg(theme::ACCENT).bold())
            .style(Style::default().bg(theme::OVERLAY_BG));

        let inner = block.inner(popup_area);
        frame.render_widget(block, popup_area);

        // Three zones: filter input, namespace list, hint bar.
        let chunks = Layout::default()
            .direction(Direction::Vertical)
            .constraints([Constraint::Length(2), Constraint::Min(1), Constraint::Length(1)])
            .split(inner);

        // Filter input — shows placeholder when empty.
        let filter_display =
            if self.filter.is_empty() { "Type to filter...".to_string() } else { self.filter.to_string() };
        let filter_style = if self.filter.is_empty() {
            Style::default().fg(theme::TEXT_DIM)
        } else {
            Style::default().fg(theme::HEADER_FG)
        };
        let filter_line = Paragraph::new(format!(" > {filter_display}")).style(filter_style);
        frame.render_widget(filter_line, chunks[0]);

        // Namespace list — "All Namespaces" is styled differently (italic, accent color).
        let filtered = self.filtered_namespaces();
        let items: Vec<ListItem> = filtered
            .iter()
            .map(|ns| {
                let style = if *ns == "All Namespaces" {
                    Style::default().fg(theme::ACCENT).italic()
                } else {
                    Style::default().fg(theme::HEADER_FG)
                };
                ListItem::new(format!("  {ns}")).style(style)
            })
            .collect();

        let list = List::new(items).highlight_style(Style::default().bg(theme::SELECTION_BG).bold());

        let mut list_state =
            ListState::default().with_selected(Some(self.selected.min(filtered.len().saturating_sub(1))));
        frame.render_stateful_widget(list, chunks[1], &mut list_state);

        let hints = Paragraph::new(" Enter:select  Esc:cancel").style(Style::default().fg(theme::TEXT_DIM));
        frame.render_widget(hints, chunks[2]);
    }
}
```
